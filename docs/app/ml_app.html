BigData & Tools USASpending Application

Module Overview

This module implements a BigData application for interacting with USASpending data. It provides various endpoints for data visualization, machine learning inference, and data management using big data tools such as Apache Spark and Cassandra.

Features:

  • Dashboard: Displays award amounts by recipient and sub-agency.
  • Machine Learning Models: Implements regression, classification, and clustering models for award data.
  • Data Fetching & Insertion: Retrieves data from the USASpending API and inserts it into Cassandra.
  • Correlation Visualization: Visualizes correlations within the data.
  • Download History Tracking: Keeps a log of data downloads.

Dependencies:

  • Apache Spark: 3.4.4
  • Apache Cassandra: 4.1.7
  • Scala: 2.12.17
  • Python: 3.12.8
  • pytest: 8.3.4
  • OpenJDK: 11.0.25
  • Flask: 3.1.0

Operating System: Ubuntu 22.04.5 LTS

Author: Buddhi Ayesha Date: 2024-12-20

 86@app.route("/dashboard", methods=["GET"])
 87def dashboard():
 88    """
 89    Render the dashboard page with various data visualizations.
 90
 91    This route reads data from Cassandra tables related to award amounts by recipient and
 92    sub-agency, as well as data for Sankey diagrams and monthly awards. The data is
 93    processed using Spark and passed to the dashboard template for rendering.
 94
 95    Returns:
 96        A rendered HTML template for the dashboard.
 97    """

 99    df_map_recipient = (

101        .format("org.apache.spark.sql.cassandra")
102        .options(table="total_award_amount_by_recipient", keyspace="usaspending_data")
103        .load()
104        .orderBy(col("total_award_amount").desc())
105        # .limit(20)
106    )
107    map_recipient_data = [row.asDict() for row in df_map_recipient.collect()]
108
109    df_map_subagency = (
110        spark.read
111        .format("org.apache.spark.sql.cassandra")
112        .options(table="total_award_amount_by_awarding_sub_agency", keyspace="usaspending_data")
113        .load()
114        .orderBy(col("total_award_amount").desc())
115        # .limit(20)
116    )
117    map_subagency_data = [row.asDict() for row in df_map_subagency.collect()]
118
119    # Read data for Sankey - Top 15 by total_award_amount
120    df_sankey = (
121        spark.read
122        .format("org.apache.spark.sql.cassandra")
123        .options(table="awarding_sub_agency_by_recipient", keyspace="usaspending_data")
124        .load()
125        .orderBy(col("total_award_amount").desc())
126        .limit(15)
127    )
128    sankey_data = [row.asDict() for row in df_sankey.collect()]
129
130    # Read data for Pie and Line - Top 30 by total_award_amount
131    df_month = (
132        spark.read
133        .format("org.apache.spark.sql.cassandra")
134        .options(table="awarding_sub_agency_by_month", keyspace="usaspending_data")
135        .load()
136        .orderBy(col("total_award_amount").desc())
137        .limit(30)
138    )
139    month_data = [row.asDict() for row in df_month.collect()]
140
141    return render_template(
142        "dashboard.html",
143        map_recipient_data=map_recipient_data,
144        map_subagency_data=map_subagency_data,
145        sankey_data=sankey_data,
146        month_data=month_data
147    )

378@app.route("/fetch_data", methods=["POST"])
379def fetch_data():
380    """
381    Fetch data from the USASpending API and insert it into Cassandra.
382
383    This route handles the data fetching process based on user-specified frequency
384    and date range. It fetches data in batches, handles pagination, and inserts
385    records into the Cassandra database. It also records the download history.
386
387    Returns:
388        A redirect to the download page with flashed messages indicating the status.
389    """
390    frequency = request.form.get('frequency')
391    start_date_str = request.form.get('start_date')
392    end_date_str = request.form.get('end_date')
393
394    try:
395        start_date = datetime.strptime(start_date_str, '%Y-%m-%d').date()
396        end_date = datetime.strptime(end_date_str, '%Y-%m-%d').date()
397        if start_date > end_date:
398            flash("Start date must be before end date.", "danger")
399            return redirect(url_for("download_page"))
400    except ValueError:
401        flash("Invalid date format. Please use YYYY-MM-DD.", "danger")
402        return redirect(url_for("download_page"))
403
404    download_start = datetime.now()
405    insert_count = 0
406    batch_size = 2000  # Records per batch
407    page_size = 100  # Records per page as per response
408    pages_per_batch = batch_size // page_size  # Number of pages per batch
409    logger.info("Data fetching frequency: %s, Pages per batch: %d", frequency, pages_per_batch)
410
411    has_next = True
412    page = 1
413    batch_counter = 0
414
415    while has_next:
416        try:
417            payload = {
418                "filters": {
419                    "time_period": [{
420                        "start_date": start_date.strftime('%Y-%m-%d'),
421                        "end_date": end_date.strftime('%Y-%m-%d')
422                    }],
423                    "award_type_codes": ["A", "B", "C"]
424                },
425                "fields": [
426                    "Award ID",
427                    "Recipient Name",
428                    "Start Date",
429                    "End Date",
430                    "Award Amount",
431                    "Awarding Agency",
432                    "Awarding Sub Agency",
433                    "Contract Award Type",
434                    "Funding Agency",
435                    "Funding Sub Agency",
436                    "Description",
437                    "Last Modified Date",
438                    "Base Obligation Date"
439                ],
440                "limit": page_size,
441                "page": page
442            }
443
444            api_url = "https://api.usaspending.gov/api/v2/search/spending_by_award/"
445            # Make the API request
446            response = requests.post(api_url, json=payload, timeout=60)  # Increased timeout
447            response.raise_for_status()
448
449            data = response.json()
450            results = data.get("results", [])
451            page_metadata = data.get("page_metadata", {})
452            has_next = page_metadata.get("hasNext", False)
453
454            if not results:
455                break
456
457            # Insert fetched records into Cassandra
458            for award in results:
459                award_id = award.get('Award ID')
460                recipient_name = award.get('Recipient Name')
461                start_dt = award.get('Start Date')
462                end_dt = award.get('End Date')
463                award_amount = award.get('Award Amount')
464                awarding_agency = award.get('Awarding Agency')
465                awarding_sub_agency = award.get('Awarding Sub Agency')
466                contract_award_type = award.get('Contract Award Type')
467                funding_agency = award.get('Funding Agency')
468                funding_sub_agency = award.get('Funding Sub Agency')
469                description = award.get('Description')
470                last_modified_date = award.get('Last Modified Date')
471                base_obligation_date = award.get('Base Obligation Date')
472
473                # Convert dates
474                start_date_converted = to_date(start_dt)
475                end_date_converted = to_date(end_dt)
476                lmd = to_date(last_modified_date)
477                base_obligation_date_converted = to_date(base_obligation_date)
478
479                # Convert award_amount to float
480                aa = None
481                if award_amount is not None:
482                    try:
483                        aa = float(award_amount)
484                    except ValueError:
485                        aa = None
486
487                # Insert into Cassandra
488                cassandra_session.execute(
489                    """
490                    INSERT INTO awards (
491                        award_id, recipient_name, start_date, end_date, award_amount,
492                        awarding_agency, awarding_sub_agency, contract_award_type,
493                        funding_agency, funding_sub_agency, description,
494                        last_modified_date, base_obligation_date
495                    ) VALUES (
496                        %s, %s, %s, %s, %s,
497                        %s, %s, %s,
498                        %s, %s, %s,
499                        %s, %s
500                    )
501                    """,
502                    (
503                        award_id, recipient_name, start_date_converted, end_date_converted, aa,
504                        awarding_agency, awarding_sub_agency, contract_award_type,
505                        funding_agency, funding_sub_agency, description,
506                        lmd, base_obligation_date_converted
507                    )
508                )
509                insert_count += 1
510                batch_counter += 1
511
512                if batch_counter >= batch_size:
513                    batch_counter = 0
514                    logger.info(f"Inserted {insert_count} records.")
515
516                    # Wait for 3 to 5 seconds
517                    wait_time = random.randint(3, 5)
518                    logger.info(f"Waiting for {wait_time} seconds before next batch...")
519                    time.sleep(wait_time)
520            page += 1
521
522        except requests.exceptions.ConnectionError as ce:
523            logger.error(f"Connection error encountered: {ce}")
524            logger.error("Retrying after a 3 seconds delay...")
525            time.sleep(3)
526            continue
527        except requests.exceptions.Timeout as te:
528            logger.error(f"Request timed out: {te}")
529            logger.error("Retrying after a 3 seconds delay...")
530            time.sleep(3)
531            continue
532        except requests.exceptions.HTTPError as he:
533            logger.error(f"HTTP error occurred: {he}")
534            flash(f"HTTP error: {he}", "danger")
535            return redirect(url_for("download_page"))
536        except Exception as e:
537            logger.error(f"An unexpected error occurred: {e}")
538            flash(f"An unexpected error occurred: {e}", "danger")
539            return redirect(url_for("download_page"))
540
541    # Record download metadata
542    download_end = datetime.now()
543    download_id = uuid.uuid4()
544    try:
545        cassandra_session.execute(
546            """
547            INSERT INTO download_history (download_id, start_time, end_time, number_of_rows)
548            VALUES (%s, %s, %s, %s)
549            """,
550            (download_id, download_start, download_end, insert_count)
551        )
552    except Exception as e:
553        logger.error(f"Error recording download history: {e}")
554        flash("Data fetched but failed to record download history.", "warning")
555        return redirect(url_for("download_page"))
556
557    flash(f"Fetched and inserted records into Cassandra. Download log recorded.", "success")
558    return redirect(url_for("download_page"))

561def to_date(d):
562    """
563    Convert a string to a date object.
564
565    Args:
566        d (str): Date string in the format 'YYYY-MM-DD'.
567
568    Returns:
569        datetime.date or None: Converted date object if successful, else None.
570    """
571    if d:
572        try:
573            return datetime.strptime(d, '%Y-%m-%d').date()
574        except ValueError:
575            return None
576    return None

403        return redirect(url_for("download_page"))

409    pages_per_batch = batch_size // page_size  # Number of pages per batch
410    logger.info("Data fetching frequency: %s, Pages per batch: %d", frequency, pages_per_batch)
411
412    has_next = True
413    page = 1
414    batch_counter = 0
415
416    while has_next:
417        try:
418            payload = {

485                    except ValueError:

492                        award_id, recipient_name, start_date, end_date, award_amount,
493                        awarding_agency, awarding_sub_agency, contract_award_type,
494                        funding_agency, funding_sub_agency, description,
495                        last_modified_date, base_obligation_date

498                        %s, %s, %s,
499                        %s, %s, %s,
500                        %s, %s
501                    )
502                    """,

506                        funding_agency, funding_sub_agency, description,
507                        lmd, base_obligation_date_converted
508                    )

511                batch_counter += 1
512
513                if batch_counter >= batch_size:
514                    batch_counter = 0
515                    logger.info(f"Inserted {insert_count} records.")
516
517                    # Wait for 3 to 5 seconds
518                    wait_time = random.randint(3, 5)
519                    logger.info(f"Waiting for {wait_time} seconds before next batch...")
