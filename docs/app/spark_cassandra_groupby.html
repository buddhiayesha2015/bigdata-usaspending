BigData & Tools USASpending Application

Module Overview

This module performs data aggregation on the USASpending data stored in Cassandra. It reads the "awards" table, groups the data by awarding_sub_agency and recipient_name, sums the award_amount, and writes the aggregated results to a new Cassandra table awarding_sub_agency_by_recipient.

Dependencies:

  • pyspark

Usage: Run the module as a standalone script to execute the data processing pipeline.

25from pyspark.sql.functions import sum as _sum

27from app import config

30def awarding_agency_by_recipient():
31    """
32    This function to aggregate award amounts by awarding sub-agency and recipient.
33
34    This function initializes a Spark session with Cassandra configurations,
35    reads data from the "awards" table in the "usaspending_data" keyspace,
36    performs a group by operation on `awarding_sub_agency` and `recipient_name`,
37    sums the `award_amount`, and writes the aggregated data to the
38    "awarding_sub_agency_by_recipient" table in Cassandra.
39
40    Steps:
41        1. Initialize SparkSession with Cassandra connection settings.
42        2. Load data from the "awards" table.
43        3. Group data by `awarding_sub_agency` and `recipient_name`.
44        4. Aggregate the `award_amount` using sum.
45        5. Write the aggregated data to the new Cassandra table.
46        6. Stop the Spark session.
47
48    Raises:
49        Any exceptions raised during Spark operations will propagate upwards.
50    """
51    spark = (
52        SparkSession.builder
53        .appName("GroupByASAAndRecipient")
54        .config("spark.cassandra.connection.host", config.CASSANDRA_HOST)
55        .config("spark.cassandra.connection.port", config.CASSANDRA_PORT)
56        .getOrCreate()
57    )
58
59    # Read from the "awards" table
60    df_awards = (
61        spark.read
62        .format("org.apache.spark.sql.cassandra")
63        .options(table="awards", keyspace="usaspending_data")
64        .load()
65    )
66
67    # Group by awarding_sub_agency AND recipient_name, then sum the award_amount
68    df_grouped = (
69        df_awards
70        .groupBy("awarding_sub_agency", "recipient_name")
71        .agg(_sum("award_amount").alias("total_award_amount"))
72    )
73
74    # Write results to the new Cassandra table: awarding_sub_agency_by_recipient
75    (
76        df_grouped
77        .write
78        .format("org.apache.spark.sql.cassandra")
79        .options(table="awarding_sub_agency_by_recipient", keyspace="usaspending_data")
80        .mode("append")
81        .save()
82    )
83
84    spark.stop()
85
86
87if __name__ == "__main__":
88    awarding_agency_by_recipient()
